28  function : void generate GF()
29  from the irreducible polynomial
30  p(X) in polynomialCoeffs[0]..polynomialCoeffs[mm]
31  lookup tables: index->polynomial form alpha_to[] contains j=alpha^i;

36 void rs::generate_gf()
37 {
38  int mask = 1 ;
39  alpha_to[kGFAlpha] = 0 ;
40  for (int i=0; i<kGFAlpha; i++)
41  {
42  alpha_to[i] = mask ;
43  index_of[alpha_to[i]] = i ;
44  if (polynomialCoeffs[i])
45  alpha_to[kGFAlpha] ^= mask ;
46  mask <<= 1 ;
47  }
48  index_of[alpha_to[kGFAlpha]] = kGFAlpha ;
49  mask >>= 1 ;
50  for (int i=kGFAlpha+1; i<kCodeWordLength; i++)
51  { if (alpha_to[i-1] >= mask)
52  alpha_to[i] = alpha_to[kGFAlpha] ^ ((alpha_to[i-1]^mask)<<1) ;
53  else alpha_to[i] = alpha_to[i-1]<<1 ;
54  index_of[alpha_to[i]] = i ;
55  }
56  index_of[0] = -1 ;
57 }
58 
59 /*
60  Obtain the generator polynomial of the kNumberOfErrorsToCorrect-error correcting, length
61  kCodeWordLength=(2**mm -1) Reed Solomon code from the product of (X+alpha**i), i=1..2*kNumberOfErrorsToCorrect
62  */
63 void rs:: generator_polynomial()
64 {
65  primitive[0] = 2 ; /* primitive element alpha = 2 */
66  primitive[1] = 1 ; /* g(x) = (X+alpha) initially */
67  for (int i=2; i<=kCodeWordLength-kNumberOfDataBits; i++)
68  {
69  primitive[i] = 1 ;
70  for (int j=i-1; j>0; --j)
71  if (primitive[j] != 0)
72  primitive[j] = primitive[j-1]^ alpha_to[(index_of[primitive[j]]+i)%kCodeWordLength] ;
73  else primitive[j] = primitive[j-1] ;
74  primitive[0] = alpha_to[(index_of[primitive[0]]+i)%kCodeWordLength] ;
75  }
76 
77  for (int i=0; i<=kCodeWordLength-kNumberOfDataBits; i++) primitive[i] = index_of[primitive[i]] ;
78 }

109 int rs::decode_rs(std::vector<int> &recd)
110 {
111 
112  iLasterrorNum=0;
113  if(FormSyndromes(recd)==NO_ERRORS){
114  UpdateRecordedData(recd);
115  return NO_ERRORS;
116  }
117  int u=computeerors();
118  iLasterrorNum=l[u];
119  if (iLasterrorNum>kNumberOfErrorsToCorrect) //too much errors
120  { /* elp has degree has degree >kNumberOfErrorsToCorrect hence cannot solve */
121 
122  UpdateRecordedData(recd);
123  return TOO_MUCH_ERRORS;
124  }
125 
126  if( calculateroots(u)!=iLasterrorNum)
127  { /* elp has degree has degree >kNumberOfErrorsToCorrect hence cannot solve */
128  l[u]=10;
129  UpdateRecordedData(recd);
130  return TOO_MUCH_ERRORS;
131  }
132  /* form polynomial z(x) */
133  for (int i=1; i<=iLasterrorNum; i++) /* Z[0] = 1 always - do not need */
134  {
135  if ((s[i]!=-1) && (elp[u][i]!=-1))
136  z[i] = alpha_to[s[i]] ^ alpha_to[elp[u][i]] ;
137  else if((s[i]!=-1) && (elp[u][i]==-1))
138  z[i] = alpha_to[s[i]] ;
139  else if ((s[i]==-1) && (elp[u][i]!=-1))
140  z[i] = alpha_to[elp[u][i]] ;
141  else
142  z[i] = 0 ;
143 
144  for (int j=1; j<i; j++)
145  if ((s[j]!=-1) && (elp[u][i-j]!=-1))
146  z[i] ^= alpha_to[(elp[u][i-j] + s[j])%kCodeWordLength] ;
147  z[i] = index_of[z[i]] ; /* put into index form */
148  }
149 
150  /* evaluate errors at locations given by error location numbers loc[i] */
151  for (int i=0; i<kCodeWordLength; i++)
152  {
153  err[i] = 0 ;
154  if (recd[i]!=-1) /* convert recd[] to polynomial form */
155  recd[i] = alpha_to[recd[i]] ;
156  else recd[i] = 0 ;
157  }
158  for (int i=0; i<l[u]; i++) /* compute numerator of error term first */
159  {
160  int error=1;
161  for (int j=1; j<=l[u]; j++)
162  if (z[j]!=-1)
163  error ^= alpha_to[(z[j]+j*root[i])%kCodeWordLength] ;
164 
165 
166  if(error!=0)
167  {
168  error=index_of[error];
169  int q = 0 ; /* form denominator of error term */
170  for (int j=0; j<l[u]; j++)
171  {
172  if (j!=i)
173  q += index_of[1^alpha_to[(loc[j]+root[i])%kCodeWordLength]] ;
174  }
175  q = q % kCodeWordLength ;
176  error = alpha_to[(error-q+kCodeWordLength)%kCodeWordLength] ;
177  recd[loc[i]] ^= error ; /*recd[i] must be in polynomial form */
178 
179 
180  }
181  }
182 
183 
184  return ERRORS_CORRECTED;
185 }

198 int rs::FormSyndromes(std::vector<int> &recd)
199 {
200  int syn_error=0;
201  for (int i=1; i<=kCodeWordLength-kNumberOfDataBits; i++){
202  s[i] = 0 ;
203  for (int j=0; j<kCodeWordLength; j++)
204  if (recd[j]!=-1)
205  s[i] ^= alpha_to[(recd[j]+i*j)%kCodeWordLength] ; /* recd[j] in index form */
206  /* convert syndrome from polynomial form to index form */
207  if (s[i]!=0) syn_error=1 ; /* set flag if non-zero syndrome => error */
208  s[i] = index_of[s[i]] ;
209  }
210  return syn_error;

212 /* compute the error location polynomial via the Berlekamp iterative algorithm,
213  following the terminology of Lin and Costello : d[u] is the 'mu'th
214  discrepancy, where u='mu'+1 and 'mu' (the Greek letter!) is the step number
215  ranging from -1 to 2*kNumberOfErrorsToCorrect (see L&C), l[u] is the
216  degree of the elp at that step, and u_l[u] is the difference between the
217  step number and the degree of the elp.
218  */

221 int rs::computeerors()
222 {
223 
224  int j;
225  d[0] = 0 ; /* index form */
226  d[1] = s[1] ; /* index form */
227  elp[0][0] = 0 ; /* index form */
228  elp[1][0] = 1 ; /* polynomial form */
229  for (int i=1; i<kCodeWordLength-kNumberOfDataBits; i++)
230  {
231  elp[0][i] = -1 ; /* index form */
232  elp[1][i] = 0 ; /* polynomial form */
233  }
234  l[0] = 0 ;
235  l[1] = 0 ;
236  u_lu[0] = -1 ;
237  u_lu[1] = 0 ;
238  int u = 0 ;
239 
240  do
241  {
242  u++ ;
243  if (d[u]==-1)
244  { l[u+1] = l[u] ;
245  for (int i=0; i<=l[u]; i++)
246  { elp[u+1][i] = elp[u][i] ;
247  elp[u][i] = index_of[elp[u][i]] ;
248  }
249  }
250  else
251  /* search for words with greatest u_lu[q] for which d[q]!=0 */
252  {
253  int q = u-1 ;
254  while ((d[q]==-1) && (q>0)) q-- ;
255  /* have found first non-zero d[q] */
256  if (q>0)
257  { j=q ;
258  do
259  { j-- ;
260  if ((d[j]!=-1) && (u_lu[q]<u_lu[j]))
261  q = j ;
262  }while (j>0) ;
263  } ;
264 
265  /* have now found q such that d[u]!=0 and u_lu[q] is maximum */
266  /* store degree of new elp polynomial */
267  if (l[u]>l[q]+u-q) l[u+1] = l[u] ;
268  else l[u+1] = l[q]+u-q ;
269 
270  /* form new elp(x) */
271  for (int i=0; i<kCodeWordLength-kNumberOfDataBits; i++) elp[u+1][i] = 0 ;
272  for (int i=0; i<=l[q]; i++)
273  if (elp[q][i]!=-1)
274  elp[u+1][i+u-q] = alpha_to[(d[u]+kCodeWordLength-d[q]+elp[q][i])%kCodeWordLength] ;
275  for (int i=0; i<=l[u]; i++)
276  { elp[u+1][i] ^= elp[u][i] ;
277  elp[u][i] = index_of[elp[u][i]] ; /*convert old elp value to index*/
278  }
279  }
280  u_lu[u+1] = u-l[u+1] ;
281 
282  /* form (u+1)th discrepancy */
283  if (u<kCodeWordLength-kNumberOfDataBits) /* no discrepancy computed on last iteration */
284  {
285  if (s[u+1]!=-1)
286  d[u+1] = alpha_to[s[u+1]] ;
287  else
288  d[u+1] = 0 ;
289  for (int i=1; i<=l[u+1]; i++)
290  if ((s[u+1-i]!=-1) && (elp[u+1][i]!=0))
291  d[u+1] ^= alpha_to[(s[u+1-i]+index_of[elp[u+1][i]])%kCodeWordLength] ;
292  d[u+1] = index_of[d[u+1]] ; /* put d[u+1] into index form */
293  }
294  } while ((u<kCodeWordLength-kNumberOfDataBits) && (l[u+1]<=kNumberOfErrorsToCorrect)) ;
295 
296  return ++u;
297 }

299 int rs::calculateroots(int u)
300 {
301  /* put elp into index form */
302  for (int i=0; i<=l[u]; ++i) elp[u][i] = index_of[elp[u][i]] ;
303  /* find roots of the error location polynomial */
304  for (int i=1; i<=l[u]; i++) reg[i] = elp[u][i] ;
305 
306  int count = 0 ;
307  for (int i=1; i<=kCodeWordLength; i++)
308  {
309  int q = 1 ;
310  for (int j=1; j<=l[u]; j++)
311  if (reg[j]!=-1){
312  reg[j] = (reg[j]+j)%kCodeWordLength ;
313  q ^= alpha_to[reg[j]] ;
314  }
315  if (!q) /* store root and error location number indices */
316  {
317  root[count] = i;
318  loc[count] = kCodeWordLength-i ;
319  count++ ;
320  }
321  }
322 
323  return count;
