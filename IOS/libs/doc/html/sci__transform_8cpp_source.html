21  ComplexVectorD hilbert::transform(std::vector<int16_t > &X)
22  {
23  ComplexVectorD transformed;
24  transformed.reserve(X.size());
25  for(auto i=X.begin();i<X.end();++i)
26  {
27  float Xi = (float)(*i);
28  sci_complex tmp=DelayBuffer[0]+1if*(lpfFFilterTaps[0] * Xi + rBuffer[0]);
29  transformed.push_back(tmp);
30  for (int j = 1; j < HILBERT_FIR_ORDER; ++j)
31  rBuffer[j - 1] = lpfFFilterTaps[j] * Xi + rBuffer[j];
32  for (int j = 1; j < HILBERT_DELAY; ++j)
33  DelayBuffer[j - 1] = DelayBuffer[j];
34  DelayBuffer[HILBERT_DELAY-1] = (float)Xi;
35  rBuffer[HILBERT_FIR_ORDER-1] = lpfFFilterTaps[HILBERT_FIR_ORDER] * Xi;

70  sci_vector_d kLevinsonDurbin(sci_vector_d &R,uint8_t LPC_ORDER)
71  {
72  sci_vector_d A(LPC_ORDER+1,0);
73  sci_vector_d Ap(LPC_ORDER+1,0);
74  sci_vector_d K(LPC_ORDER+1,0);
75 
76  float E = -R[1]/R[0];
77 
78  A[0] = 1.0f;
79  Ap[0] = E ;
80  float v = ( 1 - E*E) * R[0];
81 
82  for (int t = 1 ;t<LPC_ORDER; ++t)
83  {
84 
85  float Leg=0;
86  for (int j=1;j<t+1;++j)
87  Leg+=Ap[t-j] * (float)R[j];
88 
89  E = -((float)R[t+1] + Leg) / v;
90 
91  for (int j=0;j<t;++j)
92  K[j]=Ap[j];
93 
94  for (int j=0;j<t;++j)
95  Ap[t-1-j]+=E*K[j];
96 
97  Ap[t]=E;
98  v = v* ( 1 - E*E);
99 
100  }
101  for (int j=0;j<LPC_ORDER;++j)
102  A[j+1]=(float)Ap[j];
103  return A;
104  };

107  {
108 
109  sci_vector_d _r(LPC_ORDER+1,0);
110  float norm=1.0/(float)_window;

112  float _sum = 0.0f;
113  for (int _j = 0; _j <_window - _iCorrLag; ++_j)
114  {
115  float val=(float)_s[_j];
116  float val2=(float)_s[_j + _iCorrLag];
117  _sum += val * val2;
118  }
119  _r[_iCorrLag] = _sum*norm;
120  }
121  return _r;
